#Scholars from where?
#Country Scope
scholar_from_our_school_trigger = {
	has_country_modifier = $school$_scholar_modifier
	ROOT = {
		religious_school = {
			religion_group = muslim
			OR = {
				school = $school$_school
				school = $school$_school
				school = $school$_school
			}
		}
	}
}

our_scholar_matches_their_school_trigger = {
	PREV = {
		has_country_modifier = $school$_scholar_modifier
	}
	religious_school = {
		religion_group = muslim
		OR = {
			school = $school$_school
			school = $school$_school
			school = $school$_school
		}
	}
}
