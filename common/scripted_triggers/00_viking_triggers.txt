### COLONY STUFF
can_country_flip_to_colonial_culture = {
	#to
	is_culture_to_$to$ = {
		cond = primary_culture
	}
	any_owned_province = {
		is_region_to_$to$ = yes
		is_culture_to_$to$ = {
			cond = culture
		}
	}
}

can_province_flip_to_colonial_culture = {
	#to
	is_culture_to_$to$ = {
		cond = culture
	}
	is_region_to_$to$ = yes
}
