local_church = {
    icon = GFX_holy_order_dominicans
    trigger = {
        OR = {
            religion = orthodox
            religion = paulician
            religion = bogomilist
        }
    }
    cost = 100
    cost_type = adm_power
    per_province_effect = {
		tooltip = {
            custom_tooltip = orthdox_church_trading.tt
        }
        hidden_effect = {
            if = {
                limit = {
					NOT = { development = 5 }
                }
                add_base_tax = 1
				add_base_production = 1
				add_base_manpower = 1
            }
			if = {
                limit = {
					development = 5
                }
				add_prosperity = 25
            }
        }
    }
    per_province_abandon_effect = {
        add_local_autonomy = 50
    }
    modifier = {
        local_missionary_strength = 0.08
    }
    ai_priority = {
        factor = 1
        modifier = {
            factor = 0.1
            development = 5
        }
		modifier = {
            factor = 1.5
			NOT = { development = 5 }
        }
		modifier = {
            factor = 2.0
			NOT = { religion = ROOT }
        }
    }
}
