country_event = {
	id = culture_shift.10002
	title = "culture_shift.EVTNAME10002"
	desc = "culture_shift.EVTDESC10002"
	picture = COLONIZATION_eventPicture
	fire_only_once = yes
	trigger = {
		any_owned_province = {
			region = south_africa_region
			OR = {
				culture = afrikaner
				culture = rhodesian
				culture = sudafricano
				culture = sulafricano
				culture = anafraic
			}
			is_subject = no
			is_at_war = no
			NOT = {
				exists = SAA
			}
		}
	}
	mean_time_to_happen = {
		years = 10
	}
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					NOT = {
						capital_scope = {
							region = south_africa_region
						}
					}
				}
				every_owned_province = {
					limit = {
						region = south_africa_region
					}
					add_core = SAA
				}
				release = SAA
				SAA = {
					change_government = republic
					change_religion = catholic
					add_government_reform = afrikaner_parliament_reform
					adopt_reform_progress = ROOT
					capital_scope = {
						build_to_forcelimit = {
							infantry = 0.6
							cavalry = 0.1
							artillery = 0.3
						}
						if = {
							limit = {
								culture = afrikaner
							}
							PREV = {
								change_primary_culture = afrikaner
							}
						}
						else_if = {
							limit = {
								culture = rhodesian
							}
							PREV = {
								change_primary_culture = rhodesian
							}
						}
						else_if = {
							limit = {
								culture = sudafricano
							}
							PREV = {
								change_primary_culture = sudafricano
							}
						}
						else_if = {
							limit = {
								culture = sulafricano
							}
							PREV = {
								change_primary_culture = sulafricano
							}
						}
						else = {
							limit = {
								culture = anafraic
							}
							PREV = {
								change_primary_culture = anafraic
							}
						}
					}
				}
			}
		}
	}
	option = {
		name = "culture_shift.EVTOPTA10002"
		south_africa_region = {
			remove_core = ROOT
		}
		create_subject = {
			subject_type = vassal
			subject = SAA
		}
	}
	option = {
		# Reject Freedom
		name = "culture_shift.EVTOPTB10002"
		add_stability = -2
		declare_war_with_cb = {
			who = SAA
			casus_belli = cb_annex
		}
	}
}
