#3 The Imperial Diet (Catholic)
country_event = {
	id = religious_leagues.6
	title = "religious_leagues.6.t"
	desc = "religious_leagues.6.d"
	picture = RELIGIOUS_WARS_eventPicture
	major = yes
	fire_only_once = yes
	trigger = {
		is_emperor = yes
		is_at_war = no
		NOT = {
			any_known_country = {
				is_in_league_war = yes
			}
		}
		has_regency = no
		hre_religion_locked = no
		hre_religion_treaty = no
		religion = catholic
		OR = {
			AND = {
				hre_leagues_enabled = yes
				had_global_flag = {
					flag = evangelical_union_happened
					days = 10950
				}
				NOT = {
					any_known_country = {
						is_elector = yes
						OR = {
							religion = protestant
							religion = reformed
							religion = cathar
							religion = waldensian
							religion = fraticelli
							religion = lollard
							religion = anglican
							religion = hussite
						}
						is_league_enemy = ROOT
						truce_with = ROOT
					}
				}
			}
			AND = {
				NOT = {
					has_global_flag = evangelical_union_happened
				}
				is_year = 1625
			}
		}
	}
	mean_time_to_happen = {
		months = 60
	}
	option = {
		name = "religious_leagues.6.a"
		set_hre_religion_locked = yes
	}
}

#2 Victory of the Protestant League
country_event = {
	id = religious_leagues.2
	title = "religious_leagues.2.t"
	desc = "religious_leagues.2.d"
	picture = RELIGIOUS_WARS_eventPicture
	major = yes
	is_triggered_only = yes
	trigger = {
		hre_religion = catholic
		OR = {
			NOT = {
				has_country_flag = hre_religion_changed
			}
			had_country_flag = {
				flag = hre_religion_changed
				days = 30
			}
		}
	}
	immediate = {
		set_country_flag = hre_religion_changed
		hidden_effect = {
			#set to treaty first so we can set everything else without elections being called at the wrong time
			set_hre_religion_treaty = yes
			set_hre_heretic_religion = catholic
			set_hre_religion = protestant
			#now set it to locked once we've completed the tasks
			set_hre_religion_locked = yes
			#Check whether Dissensions among the Reformers incident should happen
			#Needs 2/3 of the Empire to be owned by Reformed princes
			while = {
				limit = {
					any_province = {
						is_part_of_hre = yes
						owner = {
							is_part_of_hre = yes
							religion = reformed
						}
						NOT = {
							has_province_flag = hre_post_leaguewar_reformed_owned
						}
					}
				}
				1 = {
					#Stockholm is the global variable substitute
					change_variable = {
						which = reformed_hre_member_provinces
						value = 1
					}
				}
				random_province = {
					limit = {
						is_part_of_hre = yes
						owner = {
							is_part_of_hre = yes
							religion = reformed
						}
						NOT = {
							has_province_flag = hre_post_leaguewar_reformed_owned
						}
					}
					set_province_flag = hre_post_leaguewar_reformed_owned
				}
			}
			while = {
				limit = {
					any_province = {
						is_part_of_hre = yes
						owner = {
							is_part_of_hre = yes
							NOT = {
								religion = reformed
							}
						}
						NOT = {
							has_province_flag = hre_post_leaguewar_non_reformed_owned
						}
					}
				}
				1 = {
					change_variable = {
						which = non_reformed_hre_member_provinces
						value = 1
					}
				}
				random_province = {
					limit = {
						is_part_of_hre = yes
						owner = {
							is_part_of_hre = yes
							NOT = {
								religion = reformed
							}
						}
						NOT = {
							has_province_flag = hre_post_leaguewar_non_reformed_owned
						}
					}
					set_province_flag = hre_post_leaguewar_non_reformed_owned
				}
			}
			every_province = {
				limit = {
					is_part_of_hre = yes
				}
				clr_province_flag = hre_post_leaguewar_non_reformed_owned
				clr_province_flag = hre_post_leaguewar_reformed_owned
			}
			1 = {
				set_variable = {
					which = reformed_others_member_provinces_ratio
					which = reformed_hre_member_provinces
				}
				divide_variable = {
					which = reformed_others_member_provinces_ratio
					which = non_reformed_hre_member_provinces
				}
				if = {
					limit = {
						check_variable = {
							which = reformed_others_member_provinces_ratio
							value = 1.1
						}
					}
					set_global_flag = incident_reformer_dissension_can_happen
				}
				if = {
					limit = {
						check_variable = {
							which = reformed_others_member_provinces_ratio
							value = 2
						}
					}
					set_global_flag = incident_reformer_dissension_high_chance
				}
			}
		}
	}
	option = {
		name = "religious_leagues.2.a"
		tooltip = {
			set_hre_heretic_religion = catholic
			set_hre_religion = protestant
			set_hre_religion_locked = yes
		}
		if = {
			limit = {
				has_dlc = "Emperor"
				has_global_flag = incident_reformer_dissension_can_happen
			}
			custom_tooltip = incident_reformer_dissension_can_happen_tt
			set_imperial_incident = incident_reformer_dissension
		}
	}
	after = {
		1 = {
			set_variable = {
				which = non_reformed_hre_member_provinces
				value = 0
			}
			set_variable = {
				which = reformed_hre_member_provinces
				value = 0
			}
			set_variable = {
				which = reformed_others_member_provinces_ratio
				value = 0
			}
		}
	}
}
